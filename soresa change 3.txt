package com.day5.looping;

public class LoopStmtsDemo3 {

	public static void main(String[] args) {
		
//		write java code that will print the sum/total of numbers from 1 to 20.
		
		int sum = 0;

		for(int j=1; j<=20; j++) {
			
			sum = sum + j;
			
		}
		
		System.out.println("The total: " + sum);
		

	}

}
